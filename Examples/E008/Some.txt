package Examples.E008;

public class EncapsulationSettersAndGetters {
    public static void main(String[] args) {
        Person person1 = new Person();
        person1.setName("Tom");
        person1.age = 5;
        //person1.getInfo() //если будет нужно sout name + age
        person1.speak();

    }
}

    String name;
    int age;

    public void setName(String userName){
        if (userName.isEmpty()){
            System.out.println("Вы ввели пустое имя");
        }else {
        name = userName;}
    }

    int calculaterYearsToRetirement(){
        int years = 65-age;
        return years;
    }

    void speak(){
        for(int i = 0; i<3; i++){
            System.out.println("Меня зовут "+ name +",мне"+age+" ,лет");
        }
    }
}
